// ====================== // CSV 파일 불러오기 // ====================== Papa.parse("서울시 에코마일리지 인센티브 상품 정보.csv",{ download:true, header:true, skipEmptyLines:true, complete:function(result){ data=result.data; filtered=[...data]; initialize(); } }); // ====================== // 초기화 // ====================== function initialize(){ makeCards(); makeFilter(); makeTable(filtered); drawCharts(); } // ====================== // 가격 숫자로 변환 // ====================== function getPrice(value){ if(!value) return 0; return Number( value .toString() .replace(/,/g,"") .replace(/원/g,"") .replace(/[^\d]/g,"") ); } // ====================== // 통계 카드 // ====================== function makeCards(){ document.getElementById("totalCount").innerText=data.length; let prices=data.map(d=>getPrice(d["상품가격"])); let avg=Math.round( prices.reduce((a,b)=>a+b,0) /prices.length ); let max=Math.max(...prices); document.getElementById("avgPrice").innerText= avg.toLocaleString()+"원"; document.getElementById("maxPrice").innerText= max.toLocaleString()+"원"; let brands= new Set( data.map(d=>d["브랜드/제조사 명칭"]) ); document.getElementById("brandCount").innerText= brands.size; } // ====================== // 필터 생성 // ====================== function makeFilter(){ let set=new Set(); data.forEach(d=>{ set.add(d["상품타입"]); }); let select=document.getElementById("typeFilter"); [...set].sort().forEach(type=>{ let op=document.createElement("option"); op.value=type; op.innerText=type; select.appendChild(op); }); } // ====================== // 테이블 출력 // ====================== function makeTable(list){ let tbody=document.getElementById("tableBody"); tbody.innerHTML=""; list.forEach(d=>{ tbody.innerHTML+=` ${d["상품명"]} ${d["상품가격"]} ${d["상품타입"]} ${d["브랜드/제조사 명칭"]} ${d["배송여부"]} `; }); } // ====================== // 검색 // ====================== document.getElementById("search") .addEventListener("keyup",filterData); // ====================== // 타입 필터 // ====================== document.getElementById("typeFilter") .addEventListener("change",filterData); function filterData(){ let keyword= document .getElementById("search") .value .trim(); let type= document .getElementById("typeFilter") .value; filtered=data.filter(d=>{ let ok1=true; let ok2=true; if(keyword!=""){ ok1=d["상품명"] .includes(keyword); } if(type!=""){ ok2=d["상품타입"]==type; } return ok1&&ok2; }); makeTable(filtered); drawCharts(); } // ====================== // 가격 오름차순 // ====================== document .getElementById("sortLow") .onclick=function(){ filtered.sort( (a,b)=> getPrice(a["상품가격"]) - getPrice(b["상품가격"]) ); makeTable(filtered); }; // ====================== // 가격 내림차순 // ====================== document .getElementById("sortHigh") .onclick=function(){ filtered.sort( (a,b)=> getPrice(b["상품가격"]) - getPrice(a["상품가격"]) ); makeTable(filtered); }; // =============================== // Chart 객체 // =============================== let typeChart; let brandChart; let deliveryChart; let priceChart; // =============================== // 차트 생성 // =============================== function drawCharts(){ drawTypeChart(); drawBrandChart(); drawDeliveryChart(); drawPriceChart(); } // =============================== // 상품타입별 그래프 // =============================== function drawTypeChart(){ let count={}; filtered.forEach(item=>{ let key=item["상품타입"] || "기타"; count[key]=(count[key]||0)+1; }); if(typeChart) typeChart.destroy(); typeChart=new Chart( document.getElementById("typeChart"), { type:"bar", data:{ labels:Object.keys(count), datasets:[{ label:"상품 개수", data:Object.values(count), backgroundColor:"#3498db" }] }, options:{ responsive:true, plugins:{ legend:{display:false} } } } ); } // =============================== // 브랜드 TOP10 // =============================== function drawBrandChart(){ let count={}; filtered.forEach(item=>{ let key=item["브랜드/제조사 명칭"] || "기타"; count[key]=(count[key]||0)+1; }); let arr=Object.entries(count) .sort((a,b)=>b[1]-a[1]) .slice(0,10); if(brandChart) brandChart.destroy(); brandChart=new Chart( document.getElementById("brandChart"), { type:"bar", data:{ labels:arr.map(i=>i[0]), datasets:[{ label:"상품 수", data:arr.map(i=>i[1]), backgroundColor:"#2ecc71" }] }, options:{ responsive:true, indexAxis:"y" } } ); } // =============================== // 배송여부 원그래프 // =============================== function drawDeliveryChart(){ let count={}; filtered.forEach(item=>{ let key=item["배송여부"] || "기타"; count[key]=(count[key]||0)+1; }); if(deliveryChart) deliveryChart.destroy(); deliveryChart=new Chart( document.getElementById("deliveryChart"), { type:"pie", data:{ labels:Object.keys(count), datasets:[{ data:Object.values(count) }] }, options:{ responsive:true } } ); } // =============================== // 가격 분포 // =============================== function drawPriceChart(){ let bucket={ "1만원 이하":0, "1~3만원":0, "3~5만원":0, "5~10만원":0, "10만원 이상":0 }; filtered.forEach(item=>{ let p=getPrice(item["상품가격"]); if(p<=10000){ bucket["1만원 이하"]++; } else if(p<=30000){ bucket["1~3만원"]++; } else if(p<=50000){ bucket["3~5만원"]++; } else if(p<=100000){ bucket["5~10만원"]++; } else{ bucket["10만원 이상"]++; } }); if(priceChart) priceChart.destroy(); priceChart=new Chart( document.getElementById("priceChart"), { type:"bar", data:{ labels:Object.keys(bucket), datasets:[{ label:"상품 수", data:Object.values(bucket), backgroundColor:"#f39c12" }] }, options:{ responsive:true, plugins:{ legend:{display:false} } } } ); } // =============================== // 필터 적용 시 차트 갱신 // =============================== function refresh(){ makeTable(filtered); drawCharts(); } document.getElementById("search").addEventListener("keyup",refresh); document.getElementById("typeFilter").addEventListener("change",refresh);
